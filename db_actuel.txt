    LANGUAGE plpgsql
    AS $$

BEGIN

  IF (TG_OP = 'DELETE' OR TG_OP = 'UPDATE') THEN

    IF EXISTS (SELECT 1 FROM subcategories WHERE category_id = OLD.id) THEN

      RAISE EXCEPTION 'Operation % disallowed: category % has linked subcategories', TG_OP, OLD.id;

    END IF;

  END IF;

  IF TG_OP = 'DELETE' THEN

  END IF;

END;
